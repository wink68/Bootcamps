```
  </script>

```

<br>

__JS >__
```
function checkSign(num) {
    let numString = num.toString();
    
    if (numString[0] === "-") {
        numString = "negative";
    } else if (numString[0] === "0") {
        numString = "zero";
    } else if (numString[0] !== "-" && numString[0] !== "0") {
        numString = "positive";
    }
    return numString;
}
```
