     var reverStr = '';     // 빈 문자열

     for (var i = str.length -1; i >= 0; i--) {    // str의 끝자리 인덱스부터
        reverStr = reverStr + str.charAt(i);      // 빈 문자열에 점차 붙여나감
     } return reverStr;
  }

### 7) 소수 구하기
__ex1>__: n까지 소수의 합   
```
  function isPrime(n) {
     var sum = 0;
     
     for (var i=1; i <= n; i++) {     // n까지 수
        for (var j=2; j <= i; j++) {  // 약수
           if (j == i) { sum += j; }
           
           if (i % j == 0) { break;}
        }
     }
     return sum;
  }
```
<br>

__ex2>__: 약수인지 여부   
```
  function isPrime(n) {
     
     for (var i=2; i <= n; i++) {     // 약수
           if (n == i) { return true; }
           
           if (n % i == 0) { return false;}
     }
  }
```
